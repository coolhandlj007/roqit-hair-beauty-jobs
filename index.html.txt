    // --- Utilities ---
    const £ = v => new Intl.NumberFormat('en-GB', { style:'currency', currency:'GBP', maximumFractionDigits:0 }).format(v);

    function jobCard(j){
      const badges = [
        j.urgent ? '<span class="badge urgent">URGENT</span>' : '',
        j.isNew ? '<span class="badge new">NEW</span>' : ''
      ].join('');
      const tags = j.tags.map(t=>`<span class="tag">${t}</span>`).join('');
      return `
        <article class="job" data-title="${j.title}" data-company="${j.company}" data-city="${j.city}" data-type="${j.type}" data-min="${j.salaryMin}" data-max="${j.salaryMax}">
          <a href="/jobs/${j.id}" aria-label="${j.title} at ${j.company}">
            <img src="${j.image}" loading="lazy" alt="${j.company} job image">
          </a>
          <div class="job-body">
            <div class="job-head">
              <div>
                <h3 class="job-title"><a class="link" href="/jobs/${j.id}">${j.title}</a></h3>
                <p class="job-company">${j.company}</p>
              </div>
              <div class="badges">${badges}</div>
            </div>
            <div class="meta">
              <div><i class="fas fa-map-marker-alt" aria-hidden="true"></i> <span>${j.city}</span></div>
              <div class="salary">${£(j.salaryMin)} – ${£(j.salaryMax)}</div>
            </div>
            <div class="tags">${tags}</div>
            <div class="job-actions">
              <a class="btn btn-primary link" href="/jobs/${j.id}">View</a>
              <a class="btn btn-ghost link" href="/apply/${j.id}" aria-label="Apply for ${j.title}">Apply</a>
            </div>
          </div>
        </article>
      `;
    }

    function render(list){
      document.getElementById('jobs').innerHTML = list.map(jobCard).join('');
      // JSON-LD for SEO (simplified per item)
      const items = list.map(j => ({
        "@context":"https://schema.org",
        "@type":"JobPosting",
        "title": j.title,
        "hiringOrganization": { "@type":"Organization", "name": j.company },
        "employmentType": j.type,
        "jobLocation": { "@type":"Place", "address": { "@type":"PostalAddress", "addressLocality": j.city, "addressCountry":"GB" } },
        "baseSalary": { "@type":"MonetaryAmount", "currency":"GBP", "value": { "@type":"QuantitativeValue", "minValue": j.salaryMin, "maxValue": j.salaryMax, "unitText":"YEAR" } },
        "applicantLocationRequirements": { "@type":"Country","name":"United Kingdom"}
      }));
      document.getElementById('ldJobs').textContent = JSON.stringify(items);
    }

    // --- Filtering / Search ---
    function applyFilters(e){
      e?.preventDefault();
      const q = (document.getElementById('q').value || '').toLowerCase().trim();
      const role = document.getElementById('f-role').value;
      const loc  = document.getElementById('f-location').value;
      const type = document.getElementById('f-type').value;
      const min  = parseInt(document.getElementById('minPay').value || '0', 10);
      const max  = parseInt(document.getElementById('maxPay').value || '9999999', 10);

      const out = JOBS.filter(j=>{
        const hay = (j.title+' '+j.company+' '+j.city+' '+j.tags.join(' ')).toLowerCase();
        const matchesQ = !q || hay.includes(q);
        const matchesRole = !role || j.title === role;
        const matchesLoc  = !loc || j.city === loc;
        const matchesType = !type || j.type === type;
        const matchesPay  = (j.salaryMin >= min) && (j.salaryMax <= max);
        return matchesQ && matchesRole && matchesLoc && matchesType && matchesPay;
      });
      render(out);
    }

    document.getElementById('searchForm').addEventListener('submit', applyFilters);
    document.getElementById('filterForm').addEventListener('submit', applyFilters);
    document.getElementById('q').addEventListener('input', ()=>applyFilters());

    // Initial render
    render(JOBS);
